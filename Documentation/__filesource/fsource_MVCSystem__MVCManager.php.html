  116.          * Pre-fetch
  117.          */

  119.         if( $UriPath !== null ) {

  121.             //Event::Debug( $_REQUEST );
  122.             $SeoRequest = $_REQUEST;
  123.             $UriPath = Seo::Request( $UriPath );
  124.             //Event::Message( 'After: '.$UriPath,__METHOD__ );
  125.             //Event::Debug( $_REQUEST );
  126.         }
  127.         /**
  128.          * Fetch route
  129.          */
  130.         $MVCRoute = MVCRouter::Route( $UriPath );
  131.         if(self::DEBUG){Event::Message('Route: ['.$MVCRoute->optionRoute().']',__METHOD__,__LINE__);};
  132.         /**
  133.          * Check route restriction
  134.          */
  135.         if( $MVCRoute->IsRestricted() && Auth::IsValid() ) {
  136.             if(self::DEBUG){Event::Message('Restricted User');}
  137.             $MVCRight = str_replace('\\','-',$MVCRoute->GetController().'-'.$MVCRoute->GetAction());
  138.             /**
  139.              * Create/Edit access right
  140.              */
  141.             Auth::EditRight( $MVCRight,'created by MVC-System',$MVCRoute->optionSource() );
  142.             /**
  143.              * Check access right
  144.              */
  145.             if( !Auth::Checkpoint( $MVCRight,$MVCRoute->optionSource() ) ) {
  146.                 // No access granted and guards alerted.. "security breach from known user, GET HIM OUT!"
  147.                 if( ($UriPath === null) || ($UriPath !== null && $isDenied) ) {
  148.                     Event::Journal( 'Access to restricted route: '.$MVCRoute->optionRoute(), __CLASS__ );
  149.                     if(self::DEBUG){Event::Message( 'Access to restricted route: '.$MVCRoute->optionRoute(), __CLASS__ );}
  150.                     if( $isDenied ) {
  151.                         $MVCRoute = MVCRouter::Route( self::AccessDeniedLink().'/'.str_replace('/','>',$UriPath) );
  152.                     } else {
  153.                     // = Main content -> Login page
  154.                         $MVCRoute = MVCRouter::Route( self::AccessDeniedRoute() );
  155.                     }
  156.                     //Event::Debug( $MVCRoute );
  157.                 // No access granted but no alert.. "somebody just watching the gate"
  158.                 } else {
  159.                     Event::Journal( 'Passing restricted route: '.$MVCRoute->optionRoute(), __CLASS__ );
  160.                     // = Partial content
  161.                     if(self::DEBUG){Event::Message( '[User] Passing restricted route (silent): '.$MVCRoute->optionRoute(), __CLASS__ );}
  162.                     /**
  163.                      * Post-fetch
  164.                      */
  165.                     if( $UriPath !== null ) {
  166.                         $_REQUEST = $SeoRequest;
  167.                     }
  168.                     return null;
  169.                 }
  170.             }
  171.         } else if( $MVCRoute->IsRestricted() ) {
  172.             if(self::DEBUG){Event::Message('Restricted Guest');}
  173.             // No access granted and guards alerted.. "security breach from unknown user, GET HIM OUT!"
  174.             if( ($UriPath === null) || ($UriPath !== null && $isDenied) ) {
  175.                 Event::Journal( 'Access to restricted route: '.$MVCRoute->optionRoute(), __CLASS__ );
  176.                 if(self::DEBUG){Event::Message( 'Access to restricted route: '.$MVCRoute->optionRoute(), __CLASS__ );}
  177.                 if( $isDenied ) {
  178.                     $MVCRoute = MVCRouter::Route( self::AccessDeniedLink().'/'.str_replace('/','>',$UriPath) );
  179.                 } else {
  180.                     // = Main content -> Login page
  181.                     $MVCRoute = MVCRouter::Route( self::AccessDeniedRoute() );
  182.                 }
  183.                 //Event::Debug( $MVCRoute );
  184.                 // No access granted but no alert.. "somebody just watching the gate"
  185.             } else {
  186.                 Event::Journal( 'Passing restricted route: '.$MVCRoute->optionRoute(), __CLASS__ );
  187.                 // = Partial content
  188.                 if(self::DEBUG){Event::Message( '[Guest] Passing restricted route (silent): '.$MVCRoute->optionRoute(), __CLASS__ );}
  189.                 /**
  190.                  * Post-fetch
  191.                  */
  192.                 if( $UriPath !== null ) {
  193.                     $_REQUEST = $SeoRequest;
  194.                 }
  195.                 return null;
  196.             }
  197.         } else {
  198.             //if(self::DEBUG){Event::Message('Free Access');}
  199.         }
  200.         /**
  201.          * Create controller
  202.          */
  203.         /** @var string $MVCController */
  204.         $MVCController = $MVCRoute->GetController();
  205.         /** @var MVCController $MVCController */
  206.         $MVCController = new $MVCController;
  207.         /**
  208.          * Fetch required parameters
  209.          */
  210.         /** @var \ReflectionClass $RefMVCController */
  211.         $RefMVCController = new \ReflectionClass( $MVCRoute->GetController() );
  212.         /**
  213.          * No Method available ? -> NoRoute
  214.          */
  215.         if( ! $RefMVCController->hasMethod( $MVCRoute->GetAction() ) ) {
  216.             $MVCRoute = MVCRouter::NoRoute( $MVCRoute->optionRoute() );
  217.             $MVCController = $MVCRoute->GetController();
  218.             $MVCController = new $MVCController;
  219.             $RefMVCController = new \ReflectionClass( $MVCRoute->GetController() );
  220.         }
  221.         /** @var \ReflectionMethod $RefMVCAction */
  222.         $RefMVCAction = $RefMVCController->getMethod( $MVCRoute->GetAction() );
  223.         /** @var \ReflectionParameter[] $RefMVCParameterDefinition */
  224.         $RefMVCParameterDefinition = $RefMVCAction->getParameters();
  225.         /**
  226.          * Create combined parameter array
  227.          */
  228.         $RefMVCParameterList = array();
  229.         /** @var \ReflectionParameter $RefMVCParameter */
  230.         foreach( (array)$RefMVCParameterDefinition as $RefMVCParameter ) {
  231.             if( isset( $_REQUEST[$RefMVCParameter->getName()] ) ) {
  232.                 //Event::Message('In Request? -> Request to Parameter');
  233.                 array_push( $RefMVCParameterList, $_REQUEST[$RefMVCParameter->getName()] );
  234.             } else if( in_array( $RefMVCParameter->getName(), array_keys( $MVCRoute->GetParameter() ) ) ) {
  235.                 //Event::Message('In Default? -> Default to Parameter');
  236.                 array_push( $RefMVCParameterList, $MVCRoute->GetParameter( $RefMVCParameter->getName() ) );
  237.             } else {
  238.                 //Event::Message('Empty? -> push NULL');
  239.                 array_push( $RefMVCParameterList, null );
  240.             }
  241.         }
  242.         /**
  243.          * Post-fetch
  244.          */
  245.         if( $UriPath !== null ) {
  246.             $_REQUEST = $SeoRequest;
  247.         }
  248.         /**
  249.          * Execute route
  250.          */
  251.         return $MVCController->Execute( $MVCRoute->GetAction(), $RefMVCParameterList );
  252.     }
