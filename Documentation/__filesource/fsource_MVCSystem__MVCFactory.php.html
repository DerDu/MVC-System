  55.      * This method is used to create/overwrite a full model
  56.      *
  57.      * @static
  58.      * @param string $ModelNamespace __NAMESPACE__
  59.      * @param string $ModelName 
  60.      * @param string $TableName 
  61.      * @param false|boolean$doOverwrite

  63.      * @param int $Level

  66.     public static function BuildModel( $ModelNamespace, $ModelName, $TableName, $doOverwrite = false, $Directory = null, $Level = 0 ) {
  67.         $TableName = str_replace(' ', '', ucwords(
  68.             preg_replace( array('![^\w\d]!is', '!\s{2,}!'), array(' ',''),
  69.                 self::MVCFACTORY_MODELSTORE_PREFIX.'\\'.$ModelNamespace.'\\'.$TableName
  70.             )
  71.         ));
  72.         // Create Store
  73.         $TableList = Database::ADOConnection()->MetaTables('TABLES');
  74.         $TableList = array_map( 'strtoupper', $TableList );
  75.         if( !in_array( strtoupper($TableName), $TableList ) ) {
  76.             self::CreateStore( $TableName );
  77.         } else if( true === $doOverwrite ) {
  78.             Database::DropTable( $TableName );
  79.             self::CreateStore( $TableName );
  80.         }
  81.         // Create Model
  82.         self::CreateModel( $ModelNamespace, $ModelName, $TableName, $Directory, $Level );
  83.     }

  85.     /**
  86.      * Create model class
  87.      *
  88.      * @static
  89.      * @param string $Namespace __NAMESPACE__
  90.      * @param string $Name 
  91.      * @param string $Table 
  92.      * @param null|string$Directory 
  93.      * @param int $Level 
  94.      * @return void 
  95.      */
  96.     private static function CreateModel( $Namespace, $Name, $Table, $Directory = null, $Level = 0 ) {
  97.         global $ADODB_ASSOC_CASE;
  98.         if( $Directory === null ) {
  99.             $Directory = System::DirectoryCurrent();
  100.         }
  101.         /** @var \ADORecordSet $RecordSet */
  102.         $RecordSet = Database::ADOConnection()->SelectLimit("SELECT * FROM ".$Table,1);
  103.         /** @var array $Record */
  104.         $Record = $RecordSet->GetRowAssoc($ADODB_ASSOC_CASE);
  105.         $Blueprint = Template::Load( System::DirectorySyntax( __DIR__.DIRECTORY_SEPARATOR.'Factory'.DIRECTORY_SEPARATOR.'Model.tpl' ), false );
  106.         $Blueprint->Assign('ModelNamespace',$Namespace);
  107.         $Blueprint->Assign('ModelName',$Name);
  108.         $Blueprint->Assign('ModelTable',$Table);
  109.         $PropertyList = array();
  110.         $PropertyCount = 0;
  111.         foreach( (array)$Record as $PropertyName => $Void ) {
  112.             $PropertyList[$PropertyCount]['PropertyName'] = $PropertyName;
  113.             $PropertyList[$PropertyCount]['PropertyValue'] = 'null';
  114.             $PropertyCount++;
  115.         }
  116.         $Blueprint->Repeat('ModelProperty', $PropertyList );
  117.         $Blueprint->Repeat('ModelPropertyMethod', $PropertyList );
  118.         //$Model = System::File( System::CreateDirectory( System::DirectorySyntax( __DIR__.DIRECTORY_SEPARATOR.MVCLoader::BaseDirectoryApplication().DIRECTORY_SEPARATOR.$Namespace ) ).$Name.'.php' );
  119.         //Event::Message( $Directory, __METHOD__ );
  120.         if( $Level > 0 ) {
  121.             $Namespace = explode( '\\',$Namespace );
  122.             for( $WalkLevel = $Level; $WalkLevel > 0; $WalkLevel-- ) {
  123.                 array_shift( $Namespace );
  124.             }
  125.             $Namespace = implode( '\\',$Namespace );
  126.         }
  127.         $Model = System::File( System::CreateDirectory( System::DirectorySyntax( $Directory.DIRECTORY_SEPARATOR.$Namespace ) ).$Name.'.php' );
  128.         $Model->propertyFileContent( $Blueprint->Parse() );
  129.         $Model->writeFile();
  130.         //Event::Debug( $Model->propertyFileLocation(), __FILE__,__LINE__ );
  131.     }

  153.     public static function AddProperty( $Name, $Type, $Length = null ) {
  154.         $Property = array( trim($Name), trim($Type), ($Length===null?null:trim($Length)) );
  155.         // Add options
  156.         $ParameterCount = func_num_args();
  157.         for( $Run = 3; $Run < $ParameterCount; $Run++ ) {

  161.             self::$ModelPropertyStack = Stack::Priority('\\'.__CLASS__.'::SortProperty');
  162.         }
  163.         self::$ModelPropertyStack->pushData( $Property );
  164.     }

  171.      */
  172.     public static function RemoveProperty( $Name ) {
  173.         $DataList = self::$ModelPropertyStack->listData();
  174.         foreach( (array)$DataList as $Index => $Data ) {
  175.             if( $Data[0] == trim($Name) ) {
  176.                 self::$ModelPropertyStack->removeData( $Index );
  177.                 return true;
  178.             }
  179.         }
  180.         return false;
  181.     }

  183.     /**
  184.      * This method is used to sort model properties

  191.      * @return int 
  192.      */
  193.     public static function SortProperty( $A, $B ) {
  194.         $A = $A[0]; $B = $B[0];
  195.         // "id" is always first column
  196.         if( strtoupper($A) == 'ID' ) {
  197.             return -1;
  198.         }
  199.         if( strtoupper($B) == 'ID' ) {
  200.             return 1;
  201.         }
  202.         // Sort other columns
  203.         $S = array( $A, $B );
  204.         sort( $S );
  205.         if( $S[0] != $A ) {
  206.             return 1;
  207.         } else {
  208.             return -1;
  209.         }
  210.     }
